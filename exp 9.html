# Hypothesis Testing: Checking if transaction amount differs significantly between fraud and non-fraud
fraud = df[df['fraud'] == 1]['ratio_to_median_purchase_price']
non_fraud = df[df['fraud'] == 0]['ratio_to_median_purchase_price']
t_stat, p_value = stats.ttest_ind(fraud, non_fraud)
print("\nT-test p-value:", p_value)

#Supervised Learning (Classification)
X = df.drop(columns=['fraud'])
y = df['fraud']

X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.3, random_state=42)

model = LogisticRegression()
model.fit(X_train, y_train)
y_pred = model.predict(X_test)

print("\n--- Logistic Regression Report ---\n")
print(classification_report(y_test, y_pred))

#SMOTE for Class Imbalance
smote = SMOTE(sampling_strategy='minority', random_state=42)
X_resampled, y_resampled = smote.fit_resample(X, y)

X_train_sm, X_test_sm, y_train_sm, y_test_sm = train_test_split(X_resampled, y_resampled, test_size=0.3, random_state=42)

model_sm = RandomForestClassifier()
model_sm.fit(X_train_sm, y_train_sm)
y_pred_sm = model_sm.predict(X_test_sm)
